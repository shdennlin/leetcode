* LeetCode009 Palindrome Number

Questions link: https://leetcode.com/problems/palindrome-number/

** My first solution

#+begin_src python
class Solution:
    def isPalindrome(self, x: int) -> bool:
        if x <= -2**31 or x >= 2**31 - 1:
            return False

        x = str(x)
        for i in range(len(x)):
            if x[i] == x[-i - 1]:
                continue
            return False
        return True
#+end_src

#+begin_src python
class Solution:
    def isPalindrome(self, x: int) -> bool:
        if (x >= 0):
            return x == int(str(x)[::-1])
        return False
#+end_src
