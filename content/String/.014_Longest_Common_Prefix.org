* LeetCode 014. Longest Common Prefix

Questions link:	https://leetcode.com/problems/longest-common-prefix/

#+begin_src python
class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        ret = strs[0]
        if len(strs) == 1:
            return ret

        while True:
            retC = True
            for letter in strs[1:]:
                if letter.startswith(ret):
                    continue
                else:
                    ret = ret[:-1]
                    retC = False
                    break
            if retC == True:
                return ret
#+end_src
